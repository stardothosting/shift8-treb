30/**
31 * Sanitize sensitive data for logging
32 *
33 * Recursively sanitizes array data to prevent sensitive information
34 * like API tokens and passwords from appearing in logs.
35 *
36 * @since 1.0.0
37 * @param mixed $data The data to sanitize
38 * @return mixed Sanitized data with sensitive fields redacted
39 */
40function shift8_treb_sanitize_log_data($data) {
41    if (is_array($data)) {
42        $sanitized = array();
43        foreach ($data as $key => $value) {
44            $key_lower = strtolower($key);
45            if (in_array($key_lower, array('token', 'api_token', 'password', 'pass', 'pwd', 'authorization'), true)) {
46                $sanitized[$key] = '***REDACTED***';
47            } elseif (in_array($key_lower, array('username', 'user', 'login'), true)) {
48                $sanitized[$key] = strlen($value) > 2 ? substr($value, 0, 2) . '***' : '***';
49            } elseif (is_array($value)) {
50                $sanitized[$key] = shift8_treb_sanitize_log_data($value);
51            } else {
52                $sanitized[$key] = $value;
53            }
54        }
55        return $sanitized;
56    }
57    return $data;
58}

92/**
93 * Decrypt sensitive data from storage
94 *
95 * @since 1.0.0
96 * @param string $encrypted_data The encrypted data
97 * @return string The decrypted data
98 */
99function shift8_treb_decrypt_data($encrypted_data) {
100    if (empty($encrypted_data)) {
101        return '';
102    }
103    
104    // Check if this looks like a JWT token (not encrypted)
105    if (strpos($encrypted_data, 'eyJ') === 0) {
106        // This is a plain JWT token, return as-is
107        return $encrypted_data;
108    }
109    
110    // This should be encrypted data, try to decrypt it
111    $key = wp_salt('auth');
112    $data = base64_decode($encrypted_data);
113    
114    // Check if base64 decode was successful and we have enough data
115    if ($data === false || strlen($data) < 16) {
116        return '';
117    }
118    
119    $iv = substr($data, 0, 16);
120    $encrypted = substr($data, 16);
121    
122    $decrypted = openssl_decrypt($encrypted, 'AES-256-CBC', $key, 0, $iv);
123    
124    // Return decrypted data, or empty string if decryption failed
125    return $decrypted !== false ? $decrypted : '';
126}

642     * Get coordinates for a post
643     *
644     * @since 1.2.0
645     * @param int $post_id Post ID
646     * @return array|false Coordinates array or false
647     */
648    private function get_post_coordinates($post_id) {
649        // First, check stored meta fields (from geocoding or AMPRE API)
650        $stored_lat = get_post_meta($post_id, 'shift8_treb_latitude', true);
651        $stored_lng = get_post_meta($post_id, 'shift8_treb_longitude', true);
652        
653        if (!empty($stored_lat) && !empty($stored_lng)) {
654            return array(
655                'lat' => floatval($stored_lat),
656                'lng' => floatval($stored_lng)
657            );
658        }
659        
660        // Second, try to extract coordinates from post content (legacy support)
661        $post_content = get_post_field('post_content', $post_id);
662        
663        // Look for coordinate values in various formats
664        if (preg_match('/lat:\s*([0-9.-]+)/', $post_content, $lat_matches) &&
665            preg_match('/lng:\s*([0-9.-]+)/', $post_content, $lng_matches)) {
666            return array(
667                'lat' => floatval($lat_matches[1]),
668                'lng' => floatval($lng_matches[1])
669            );
670        }
671        
672        // Look for numeric coordinate values (from %MAPLAT% and %MAPLNG% placeholders)
673        if (preg_match('/([0-9.-]+),\s*([0-9.-]+)/', $post_content, $coord_matches)) {
674            $potential_lat = floatval($coord_matches[1]);
675            $potential_lng = floatval($coord_matches[2]);
676            
677            // Basic validation for Toronto area coordinates
678            if ($potential_lat > 40 && $potential_lat < 50 && $potential_lng < -70 && $potential_lng > -90) {
679                return array(
680                    'lat' => $potential_lat,
681                    'lng' => $potential_lng
682                );
683            }
684        }
685        
686        // No valid coordinates found - return false to indicate geocoding needed
687        return false;
688    }

700            'street_name' => ''
701        );

716        return $parts;
717    }
