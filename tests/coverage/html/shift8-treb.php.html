30/**
31 * Sanitize sensitive data for logging
32 *
33 * Recursively sanitizes array data to prevent sensitive information
34 * like API tokens and passwords from appearing in logs.
35 *
36 * @since 1.0.0
37 * @param mixed $data The data to sanitize
38 * @return mixed Sanitized data with sensitive fields redacted
39 */
40function shift8_treb_sanitize_log_data($data) {
41    if (is_array($data)) {
42        $sanitized = array();
43        foreach ($data as $key => $value) {
44            $key_lower = strtolower($key);
45            if (in_array($key_lower, array('token', 'api_token', 'password', 'pass', 'pwd', 'authorization'), true)) {
46                $sanitized[$key] = '***REDACTED***';
47            } elseif (in_array($key_lower, array('username', 'user', 'login'), true)) {
48                $sanitized[$key] = strlen($value) > 2 ? substr($value, 0, 2) . '***' : '***';
49            } elseif (is_array($value)) {
50                $sanitized[$key] = shift8_treb_sanitize_log_data($value);
51            } else {
52                $sanitized[$key] = $value;
53            }
54        }
55        return $sanitized;
56    }
57    return $data;
58}

92/**
93 * Decrypt sensitive data from storage
94 *
95 * @since 1.0.0
96 * @param string $encrypted_data The encrypted data
97 * @return string The decrypted data
98 */
99function shift8_treb_decrypt_data($encrypted_data) {
100    if (empty($encrypted_data)) {
101        return '';
102    }
103    
104    // Check if this looks like a JWT token (not encrypted)
105    if (strpos($encrypted_data, 'eyJ') === 0) {
106        // This is a plain JWT token, return as-is
107        return $encrypted_data;
108    }
109    
110    // This should be encrypted data, try to decrypt it
111    $key = wp_salt('auth');
112    $data = base64_decode($encrypted_data);
113    
114    // Check if base64 decode was successful and we have enough data
115    if ($data === false || strlen($data) < 16) {
116        return '';
117    }
118    
119    $iv = substr($data, 0, 16);
120    $encrypted = substr($data, 16);
121    
122    $decrypted = openssl_decrypt($encrypted, 'AES-256-CBC', $key, 0, $iv);
123    
124    // Return decrypted data, or empty string if decryption failed
125    return $decrypted !== false ? $decrypted : '';
126}

847     * @return array|false Coordinates array or false
848     */
849    private function get_post_coordinates($post_id) {
850        // First, check stored meta fields (from geocoding or AMPRE API)
851        $stored_lat = get_post_meta($post_id, 'shift8_treb_latitude', true);
852        $stored_lng = get_post_meta($post_id, 'shift8_treb_longitude', true);
853        
854        if (!empty($stored_lat) && !empty($stored_lng)) {
855            return array(
856                'lat' => floatval($stored_lat),
857                'lng' => floatval($stored_lng)
858            );
859        }
860        
861        // Second, try to extract coordinates from post content (legacy support)
862        $post_content = get_post_field('post_content', $post_id);
863        
864        // Look for coordinate values in various formats
865        if (preg_match('/lat:\s*([0-9.-]+)/', $post_content, $lat_matches) &&
866            preg_match('/lng:\s*([0-9.-]+)/', $post_content, $lng_matches)) {
867            return array(
868                'lat' => floatval($lat_matches[1]),
869                'lng' => floatval($lng_matches[1])
870            );
871        }
872        
873        // Look for numeric coordinate values (from %MAPLAT% and %MAPLNG% placeholders)
874        if (preg_match('/([0-9.-]+),\s*([0-9.-]+)/', $post_content, $coord_matches)) {
875            $potential_lat = floatval($coord_matches[1]);
